const { ethers } = require("hardhat");
const { expect, assert } = require("chai");

// describe("SimpleStorage", () => {}) //keyword mocha recognizes
describe("Store", function () {
  let StoreFactory, ourStore; //define outside the beforeEach and initialize to nothing
  //beforeEach test, we deploy a new contract
  beforeEach(async function () {
    StoreFactory = await ethers.getContractFactory("Store");
    ourStore = await StoreFactory.deploy();
  });

  it("Should start with a favorite number of 123", async function () {
    const currentValue = await ourStore.retrieve();
    const expectedValue = "123";
    // assert
    // expect
    assert.equal(currentValue.toString(), expectedValue);
    // expect(currentValue.toString()).to.equal(expectedValue)
    //toString because it can be a very large value
  });
});
